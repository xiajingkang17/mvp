from manim import *
from typing import Any, Dict
import math

BUILDERS = {}
UPDATERS = {}

def sector_area_effect(spec: Dict[str, Any]) -> Mobject:
    """
    Builds the sector area effect for the particle trajectory.
    Visualizes the 90-degree sector in the first quadrant with a semi-transparent blue fill.
    Includes labels for the time relationship.
    """
    # 1. Parse DSL
    geom = dict(spec.get("geometry", {}) or {})
    style = dict(spec.get("style", {}) or {})
    meta = dict(spec.get("meta", {}) or {})
    
    # 2. Extract Configuration
    layout = geom.get("layout", {})
    palette = style.get("palette", {})
    sizes = style.get("sizes", {})
    opacity = style.get("opacity", {})
    
    # Geometry params
    # Fix: Handle string "R" by defaulting to a numeric value suitable for visualization
    radius_str = layout.get("radius", "R")
    try:
        radius = float(radius_str)
    except ValueError:
        radius = 2.5  # Default visual radius if "R" is passed
        
    angle_span_deg = float(layout.get("angle_span_deg", 90))
    quadrant = layout.get("quadrant", "first")
    
    # Style params
    fill_color = ManimColor(palette.get("fill_color", "#00BFFF"))
    stroke_color = ManimColor(palette.get("stroke_color", "#FFFFFF"))
    stroke_width = sizes.get("stroke_width", 2)
    fill_alpha = opacity.get("fill_alpha", 0.3)
    stroke_alpha = opacity.get("stroke_alpha", 0.8)
    
    # Meta params
    latex_items = meta.get("latex_items", [])
    
    # 3. Build Elements
    
    # --- The Sector (Arc + Radius Lines) ---
    # We use an Arc for the curved part and Lines for the radii
    # Start angle: 0 (positive x-axis), End angle: 90 (positive y-axis)
    sector_arc = Arc(
        radius=radius,
        start_angle=0,
        angle=PI / 2,
        color=stroke_color,
        stroke_width=stroke_width
    )
    
    # Radius lines
    # Line along X-axis (from Origin to (R, 0))
    r1 = Line(ORIGIN, [radius, 0, 0], color=stroke_color, stroke_width=stroke_width)
    # Line along Y-axis (from Origin to (0, R))
    r2 = Line(ORIGIN, [0, radius, 0], color=stroke_color, stroke_width=stroke_width)
    
    # Combine into a group for the sector boundary
    sector_boundary = VGroup(r1, sector_arc, r2)
    
    # Create the filled area
    # We create a VMobject that traces the path: Origin -> (R,0) -> Arc -> (0,R) -> Origin
    sector_fill = VMobject()
    sector_fill.set_points_as_corners([
        [0, 0, 0],
        [radius, 0, 0]
    ])
    # Add arc points
    sector_fill.append_points(sector_arc.points)
    # Close the loop back to origin
    sector_fill.add_line_to([0, 0, 0])
    
    sector_fill.set_fill(fill_color, opacity=fill_alpha)
    sector_fill.set_stroke(width=0) # Hide stroke for the fill object, use boundary for stroke
    
    # Group fill and boundary
    sector_group = VGroup(sector_fill, sector_boundary)
    
    # --- Labels ---
    labels = VGroup()
    
    # Position labels relative to the sector
    # t = T/4
    if len(latex_items) > 0:
        label1 = MathTex(latex_items[0], font_size=32, color=WHITE)
        # Place it somewhat inside the sector or near the arc
        label1.move_to([radius * 0.6, radius * 0.6, 0])
        labels.add(label1)
        
    # t = pi*m / 2qB
    if len(latex_items) > 1:
        label2 = MathTex(latex_items[1], font_size=32, color=WHITE)
        # Place it below the sector or to the right
        label2.next_to(sector_group, DOWN, buff=0.5)
        labels.add(label2)

    # --- Main Group ---
    # We return a VGroup containing the sector and the labels.
    # The updater will handle the animation of the sector expansion.
    main_group = VGroup(sector_group, labels)
    
    # Store initial state for the updater
    # We tag the fill and boundary so the updater can find them easily
    main_group.sector_fill = sector_fill
    main_group.sector_boundary = sector_boundary
    main_group.target_angle = PI / 2
    main_group.radius_val = radius # Store radius explicitly to avoid attribute access issues
    
    # Initialize to empty state (angle 0) for the animation
    # The fill needs to be a closed loop, so at t=0 it's a line along X axis
    sector_fill.set_points_as_corners([[0,0,0], [radius, 0, 0], [0,0,0]])
    sector_arc.become(Arc(radius=radius, start_angle=0, angle=0, color=stroke_color, stroke_width=stroke_width))
    r2.become(Line(ORIGIN, ORIGIN, color=stroke_color, stroke_width=stroke_width)) # Hide the vertical radius initially
    
    return main_group

def update_sector_expansion(mobj: Mobject, t: float, spec: Dict[str, Any]) -> None:
    """
    Updates the sector area to animate its expansion from 0 to 90 degrees.
    """
    # 1. Parse Motion Config
    motion = dict(spec.get("motion", {}) or {})
    timing = motion.get("timing", {})
    duration_s = float(timing.get("duration_s", 2.0))
    rate_func_name = timing.get("rate_func", "smooth")
    
    # 2. Calculate Progress
    # Clamp t to [0, duration_s]
    local_t = min(max(t, 0), duration_s)
    progress = local_t / duration_s
    
    # Apply rate function
    if rate_func_name == "smooth":
        alpha = smooth(progress)
    elif rate_func_name == "linear":
        alpha = progress
    else:
        alpha = smooth(progress)
        
    # 3. Update Geometry
    # Current angle
    current_angle = alpha * mobj.target_angle
    # Use the explicitly stored radius value
    radius = mobj.radius_val
    
    # Update Arc
    # mobj.sector_boundary structure: [r1 (Line), arc (Arc), r2 (Line)]
    current_arc = Arc(radius=radius, start_angle=0, angle=current_angle, color=mobj.sector_boundary[1].color)
    mobj.sector_boundary[1].become(current_arc)
    
    # Update Vertical Radius (r2)
    # End point of the arc
    end_point = current_arc.get_end()
    mobj.sector_boundary[2].put_start_and_end_on(ORIGIN, end_point)
    
    # Update Fill
    # Reconstruct the filled shape based on current arc
    # Path: Origin -> (R, 0) -> Arc Points -> Origin
    new_fill = VMobject()
    new_fill.set_points_as_corners([
        [0, 0, 0],
        [radius, 0, 0]
    ])
    new_fill.append_points(current_arc.points)
    new_fill.add_line_to([0, 0, 0])
    
    # Preserve style
    new_fill.set_fill(mobj.sector_fill.get_fill_color(), opacity=mobj.sector_fill.get_fill_opacity())
    new_fill.set_stroke(width=0)
    
    mobj.sector_fill.become(new_fill)

# Register builders and updaters
BUILDERS["sector_area_effect"] = sector_area_effect
UPDATERS["sector_area_effect"] = update_sector_expansion
