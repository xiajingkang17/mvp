def motion_scene_01(self, step_id):
    if step_id == "step_02":
        block_B = self.objects.get("block_B")
        block_A = self.objects.get("block_A")
        force_arrow_F = self.objects.get("force_arrow_F")
        
        if block_B is None or block_A is None or force_arrow_F is None:
            return []

        # Motion parameters based on physics solution
        # a_A = 3.0, a_B = 17/3, t_fall = sqrt(0.3)
        # x_B_final = 0.5 * (17/3) * 0.3 = 0.85
        # x_A_final = 0.4 + 0.5 * 3.0 * 0.3 = 0.85 (Wait, relative motion logic check)
        # Relative displacement: x_B - x_A = 0.4 (A moves left relative to B)
        # x_A_final = 0.4 + 0.5 * 3.0 * 0.3 = 0.85
        # x_B_final = 0.5 * (17/3) * 0.3 = 0.85
        # Wait, if x_A_final = x_B_final, A is at the right edge of B?
        # Initial: A at 0.4 (relative to B's left edge at 0).
        # Final: A at 0.0 (relative to B's left edge).
        # So x_A_world = x_B_world + 0.0.
        # x_A_world = 0.4 + 0.5 * 3 * t^2
        # x_B_world = 0.5 * (17/3) * t^2
        # x_B - x_A = 0.5 * (8/3) * t^2 = 0.4 -> t^2 = 0.3.
        # x_B_final = 0.5 * (17/3) * 0.3 = 0.85.
        # x_A_final = 0.4 + 0.5 * 3 * 0.3 = 0.85.
        # Yes, they meet at world coordinate 0.85.
        
        run_time = 3.0
        
        # Block B motion: 0 -> 0.85
        anim_B = block_B.animate.shift(RIGHT * 0.85).set(run_time=run_time)
        
        # Block A motion: 0.4 -> 0.85
        anim_A = block_A.animate.shift(RIGHT * 0.45).set(run_time=run_time)
        
        # Force arrow moves with Block B
        anim_F = force_arrow_F.animate.shift(RIGHT * 0.85).set(run_time=run_time)

        return [anim_B, anim_A, anim_F]
    
    return []
