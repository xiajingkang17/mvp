def motion_scene_03(self, step_id):
    if step_id == "step_02":
        diagram_system = self.objects.get("diagram_system")
        if diagram_system is None:
            return []
        
        # Retrieve the specific force arrow object from the diagram group
        # Assuming the diagram_system is a VGroup or similar container
        force_f = diagram_system.force_f
        
        # Animate the label change from f_s to f_k
        # Using a replacement animation to swap the text content
        new_label = MathTex("f_k")
        new_label.match_style(force_f.label)
        new_label.move_to(force_f.label)
        
        return [
            ReplacementTransform(force_f.label, new_label),
            Indicate(force_f, scale_factor=1.2, color=YELLOW)
        ]
    return []
