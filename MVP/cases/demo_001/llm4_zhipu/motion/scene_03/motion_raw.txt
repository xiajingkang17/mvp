def motion_scene_03(self, step_id):
    if step_id == "step_01":
        diagram_A = self.objects.get("diagram_A")
        diagram_B = self.objects.get("diagram_B")
        derivation_text = self.objects.get("derivation_text")
        
        if diagram_A is None or diagram_B is None or derivation_text is None:
            return []
            
        return [
            FadeIn(diagram_A, shift=LEFT*0.5),
            FadeIn(diagram_B, shift=RIGHT*0.5),
            Write(derivation_text)
        ]
        
    if step_id == "step_02":
        diagram_A = self.objects.get("diagram_A")
        diagram_B = self.objects.get("diagram_B")
        derivation_text = self.objects.get("derivation_text")
        
        if diagram_A is None or diagram_B is None or derivation_text is None:
            return []
            
        return [
            Transform(diagram_A, diagram_A),
            Transform(diagram_B, diagram_B),
            Transform(derivation_text, derivation_text)
        ]
        
    return []
