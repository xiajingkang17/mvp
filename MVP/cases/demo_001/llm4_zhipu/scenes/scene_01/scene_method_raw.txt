def scene_scene_01(self):
    reset_scene(self, self.objects)

    zone_subtitle = (0.05, 0.95, 0.02, 0.12)
    zone_problem = (0.05, 0.95, 0.70, 0.96)
    zone_preview = (0.05, 0.95, 0.16, 0.66)

    def step_01():
        problem_text = Text(
            "题目：质量为 M=3.0kg 的长木板 B 放在光滑水平面上，长度 L=2.0m。"
            "质量为 m=1.0kg 的物块 A 静置在木板上，距左端 l=0.40m。"
            "物块与木板间动摩擦因数 μ_k=0.30，静摩擦因数 μ_s=0.40。取 g=10m/s^2。"
            "从 t=0 起，对木板施加水平向右的恒力 F=20N。忽略空气阻力。",
            font_size=24,
            color=WHITE
        )
        place_in_zone(problem_text, zone_problem, offset=(0.0, 0.2))
        register_obj(self, self.objects, "text_block_problem", problem_text)
        self.add(problem_text)

        floor = Line(
            start=np.array([-4.0, -1.0, 0.0]),
            end=np.array([4.0, -1.0, 0.0]),
            color=GRAY,
            stroke_width=2
        )
        register_obj(self, self.objects, "floor", floor)
        self.add(floor)

        block_B = Rectangle(
            width=4.0,
            height=0.4,
            color=BLUE,
            stroke_width=4,
            fill_color=BLACK,
            fill_opacity=1.0
        ).shift(UP * 0.2)
        register_obj(self, self.objects, "block_B", block_B)
        self.add(block_B)

        block_A = Rectangle(
            width=0.8,
            height=0.4,
            color=YELLOW,
            stroke_width=4,
            fill_color=BLACK,
            fill_opacity=1.0
        ).shift(LEFT * 1.2 + UP * 0.6)
        register_obj(self, self.objects, "block_A", block_A)
        self.add(block_A)

    run_step(
        self,
        self.objects,
        "首先，我们来看一下题目描述。木板B放在光滑水平面上，物块A静置在木板上，距离左端0.4米。",
        zone_subtitle,
        ["text_block_problem", "floor", "block_B", "block_A"],
        step_01,
    )

    def step_02():
        force_arrow = Arrow(
            start=np.array([2.0, 0.4, 0.0]),
            end=np.array([3.5, 0.4, 0.0]),
            buff=0.0,
            color=RED,
            stroke_width=5,
            max_tip_length_to_length_ratio=0.2
        )
        label_f = Text("F", font_size=24, color=RED).next_to(force_arrow, UP, buff=0.05)
        force_group = VGroup(force_arrow, label_f)
        register_obj(self, self.objects, "force_arrow_F", force_group)
        self.add(force_group)

        self.motion_scene_01("step_02")

    run_step(
        self,
        self.objects,
        "从t=0时刻起，对木板施加水平向右的恒力F。由于A和B之间存在摩擦力，它们都会向右加速，但加速度不同。",
        zone_subtitle,
        ["text_block_problem", "floor", "block_B", "block_A", "force_arrow_F"],
        step_02,
    )

    def step_03():
        self.motion_scene_01("step_03")

    run_step(
        self,
        self.objects,
        "最终，物块A会从木板的左端滑落。接下来，我们将分三问逐步求解。",
        zone_subtitle,
        ["text_block_problem", "floor", "block_B", "block_A", "force_arrow_F"],
        step_03,
    )

    cleanup_scene(self, self.objects, [])
