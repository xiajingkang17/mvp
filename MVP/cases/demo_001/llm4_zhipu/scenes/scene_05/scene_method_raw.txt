def scene_scene_05(self):
    reset_scene(self, self.objects)

    zone_subtitle = (0.05, 0.95, 0.02, 0.12)
    zone_diagram = (0.05, 0.5, 0.2, 0.9)
    zone_goal = (0.55, 0.95, 0.2, 0.9)

    def step_01():
        # Title for Question 2
        text_q2 = Text("第二问：求滑落时间t与速度", font_size=32, color=WHITE)
        place_in_zone(text_q2, zone_goal, offset=(0.0, 0.35))
        register_obj(self, self.objects, "text_q2", text_q2)
        self.add(text_q2)

        # Diagram: Block A on Board B
        # Board B (Longer rectangle)
        board_B = Rectangle(width=4.0, height=0.6, color=BLUE, stroke_width=4, fill_opacity=0.1)
        board_B_label = Text("B", font_size=24, color=BLUE).next_to(board_B, DOWN, buff=0.1)
        
        # Block A (Smaller rectangle on top)
        # Position A such that its right edge is at x=0.6 (representing l from left edge of B)
        # B center is at 0. B width is 4.0, so left edge is -2.0.
        # If l=0.4, A's right edge is at -2.0 + 0.4 = -1.6.
        # Let's place A at x = -1.6 - (width_A/2).
        width_A = 1.0
        pos_A_x = -1.6 - (width_A / 2.0)
        block_A = Rectangle(width=width_A, height=0.5, color=YELLOW, stroke_width=4, fill_opacity=0.2)
        block_A.move_to([pos_A_x, 0.55, 0]) # On top of B (B height 0.6, center 0)
        block_A_label = Text("A", font_size=24, color=YELLOW).move_to(block_A.get_center())

        # Relative displacement arrow l
        # Arrow from A's right edge to B's left edge
        arrow_l_start = block_A.get_right()
        arrow_l_end = board_B.get_left()
        arrow_l = Arrow(arrow_l_start, arrow_l_end, buff=0.05, color=RED, stroke_width=4)
        label_l = MathTex("l", font_size=28, color=RED).next_to(arrow_l, UP, buff=0.1)

        diagram_group = VGroup(board_B, board_B_label, block_A, block_A_label, arrow_l, label_l)
        fit_in_zone(diagram_group, zone_diagram, width_ratio=0.9)
        register_obj(self, self.objects, "diagram_rel", diagram_group)
        self.add(diagram_group)

    run_step(
        self,
        self.objects,
        "因为B的加速度大于A的加速度，A相对于B向左运动。当A相对于B移动了0.4米时，就会从左端滑落。",
        zone_subtitle,
        ["text_q2", "diagram_rel"],
        step_01,
    )

    def step_02():
        # Formula for relative motion
        # l = 1/2 * a_rel * t^2
        formula_rel = MathTex("l", "=", "\\frac{1}{2}", "a_{rel}", "t^2", font_size=36, color=WHITE)
        layout_formula_group(formula_rel, zone_goal)
        # Shift down to avoid overlapping title
        formula_rel.shift(UP * 0.5)
        
        register_obj(self, self.objects, "formula_rel", formula_rel)
        self.add(formula_rel)

    run_step(
        self,
        self.objects,
        "利用相对运动的位移公式，l等于二分之一倍相对加速度乘以t的平方。我们可以先求出相对加速度，进而求出时间t。",
        zone_subtitle,
        ["text_q2", "diagram_rel", "formula_rel"],
        step_02,
    )

    cleanup_scene(self, self.objects, [])
